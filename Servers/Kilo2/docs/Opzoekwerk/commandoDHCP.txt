--install DHCP--
Install-WindowsFeature DHCP -IncludeManagementTools

--add dhcp scope--
Add-DhcpServerV4Scope -Name "DHCP Scope" -StartRange 192.168.1.150 -EndRange 192.168.1.200 -SubnetMask 255.255.255.0

--add dns-server, router gateway--
Set-DhcpServerV4OptionValue -DnsServer 172.18.1.66, 172.18.1.67 -Router 172.18.0.1 -ScopeId 172.18.0.0

--set lease duration--
Set-DhcpServerv4Scope -ScopeId 172.18.0.0 -LeaseDuration 1.00:00:00

--restart DHCP-service--
Restart-service dhcpserver
